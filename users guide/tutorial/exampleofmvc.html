An Example of New Section

3 فایل زیر را در مسیرهای مشخص شده ایجاد کرده و نتیجه خروجی را در مرورگر مشاهده کنید.

 

ابتدا فایلی به اسم news.php در مسیر app/controllers ایجاد کنید و کدهای زیر را درون فایل قرار دهید:

<?php
class news extends ED_Controller
{
    public function last($data = NULL)
    {
       $data = $this->load->model(array('newsModel', 'news'), 'getLatestNews');

       if (isset($data))
       {
           $this->load->view(array('newsView', 'news'), 'show', $data);
        }
    }
}
?>

 

فایل دیگری به اسم news.php در مسیر app/models ایجاد کنید و کدهای زیر را درون فایل قرار دهید:

<?php
class newsModel extends ED_Model
{
    public function getLatestNews()
    {
        return  'EasyDev 1.0v Beta was published.';
    }
}
?>

 

دوباره فایلی به اسم news.php در مسیر app/views ایجاد کنید و کدهای زیر را درون فایل قرار دهید:

<?php
class newsView extends ED_View
{
    public function show($data = NULL)
    {
        if ($data != NULL)
        {
            echo $data;
        }
    }
}
?>

 

حال آدرس زیر را در مرورگر وارد کنید و نتیجه خروجی را مشاهده کنید.

http://localhost/easydev/index.php/news/last

 

در آدرس بالا easydev اسم فولدری است که فایل های فریم ورک درون آن قرار دارند.
